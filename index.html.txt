<script>

const colors = ["Red","Blue","Green","Yellow","Purple","Orange","Pink","Cyan"];
const today = new Date();
const colorIndex = today.getDate() % colors.length;
const todayColor = colors[colorIndex];
document.getElementById("colorName").textContent = todayColor;
document.getElementById("colorName").style.color = todayColor.toLowerCase();

// Ù…Ø¹Ø±Ø¶ Ø§Ù„ØµÙˆØ±
const gallery = document.getElementById('gallery');
let images = [];

document.getElementById('imageForm').addEventListener('submit', function(e){
    e.preventDefault();
    const link = document.getElementById('imageLink').value;
    const loc = document.getElementById('location').value;
    const colorTag = document.getElementById('colorTag').value || todayColor;
    if(link && loc){
        const imgObj = {link, location: loc, color: colorTag};
        images.push(imgObj);
        displayImage(imgObj);
        addMarkerToMap(loc, imgObj);
        document.getElementById('imageLink').value = '';
        document.getElementById('location').value = '';
        document.getElementById('colorTag').value = '';
    }
});

function displayImage(imgObj){
    const img = document.createElement('img');
    img.src = imgObj.link;
    img.title = `Color: ${imgObj.color} | Location: ${imgObj.location}`;
    gallery.appendChild(img);
}

// ØªØµÙÙŠØ© Ø§Ù„ØµÙˆØ± Ø­Ø³Ø¨ Ø§Ù„Ù„ÙˆÙ†
function applyFilter(){
    const filter = document.getElementById('filterColor').value.toLowerCase();
    gallery.innerHTML = '';
    images.filter(img => img.color.toLowerCase().includes(filter))
          .forEach(displayImage);
}

// Ø®Ø±ÙŠØ·Ø© ØªÙØ§Ø¹Ù„ÙŠØ©
const map = L.map('map').setView([20,0], 2);
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '&copy; OpenStreetMap contributors'
}).addTo(map);

function addMarkerToMap(location, imgObj){
    fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${location}`)
    .then(res => res.json())
    .then(data => {
        if(data && data.length > 0){
            const lat = data[0].lat;
            const lon = data[0].lon;
            const marker = L.marker([lat, lon]).addTo(map);
            marker.bindPopup(`<img src="${imgObj.link}" width="100"><br>Color: ${imgObj.color}<br>Location: ${imgObj.location}`);
        }
    });
}
